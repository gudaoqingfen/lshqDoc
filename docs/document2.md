```
/**
 * $DOC$
 * 
 * @author $AUTHOR$
 * @date $DATE$

 * {@link ,}
 */
 
 
 methodTemplate:
 
 /**
 * $DOC$
 * 
 * $PARAMS$
 * $RETURN$
 * @author $AUTHOR$
 * @date $DATE$
 * $THROWS$
 * {@link, }
 */
 
 filedTemplat:
 
 /**     $DOC$ */
```
